异或运算即可。两两抵消，剩下一个即答案:
```cpp
class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int number = 0;
        for (int i = 0; i < nums.size();i++){
            number = nums.at(i) ^ number;
        }
        return number;
    }
};
```
